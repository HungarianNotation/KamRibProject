CREATE DATABASE KamRib
GO

USE KamRib

CREATE TABLE Products
(
	Id INT PRIMARY KEY IDENTITY,
	ProductName NVARCHAR(200) NOT NULL,
	Price MONEY NOT NULL,
	StoreIds NVARCHAR(200) NOT NULL,
)
GO

CREATE TABLE StoreNames
(
	Id INT PRIMARY KEY IDENTITY,
	StoreName NVARCHAR(200) UNIQUE NOT NULL,
)
GO

CREATE TABLE Stores
(
	Id INT PRIMARY KEY IDENTITY,
	Adress NVARCHAR(300) NOT NULL,
	StoreName NVARCHAR(200) FOREIGN KEY REFERENCES StoreNames (StoreName) NOT NULL,
)
GO

CREATE TABLE Supplier
(
	Id INT PRIMARY KEY IDENTITY,
	SupplierName NVARCHAR(200) UNIQUE NOT NULL,
	SupplierLogin NVARCHAR(30) UNIQUE NOT NULL,
	SupplierPassword NVARCHAR(30) NOT NULL,
)
